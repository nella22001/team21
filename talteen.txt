<?php
function tarkistaJson($json){
    if (empty($json)){
        return false;
    }
    $reseptit=json_decode($json, false);
    if (empty($reseptit->nimi) || empty($reseptit->ainekset) || empty($reseptit->ohje) ){
        return false;
    }
    return $reseptit;
}
